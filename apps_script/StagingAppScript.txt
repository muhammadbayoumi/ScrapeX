/* ============================================================================
 * ScrapeX — Staging Funnel (Apps Script, bound to the "ScrapeX Staging" sheet)
 * ----------------------------------------------------------------------------
 * ENGINEERING.md S1: "GAS stays dumb."
 *   - doPost = token check + signature check + append ONE chunk row + ack. Zero
 *     parsing, zero reassembly (that happens in Python at ingest, where it's
 *     testable). See "REQUEST SIGNING" below for why unsigned senders still
 *     work until the owner says otherwise.
 *   - LockService around every append (30 concurrent-execution safety).
 *   - Chunks arrive <= 40,000 chars (client-enforced), safely under the 50,000
 *     char per-cell limit.
 *   - Total logic deliberately tiny; its test suite is the shared payload
 *     contract fixtures (T8) — see scraper/contracts/.
 *
 * TABLE SYNC (second half of this file) is a READ path the owner triggers, and
 * it does not weaken S1: doPost is untouched and still appends verbatim. The
 * sync only ever READS _INBOX and writes derived, disposable tabs. If it has a
 * bug the raw inbox — the thing Python ingests — is unharmed, and rerunning it
 * repairs the tabs. It follows payload.reassemble_chunks(): group by
 * (source_key, scraped_at), order by chunk.index, and REFUSE an incomplete
 * batch rather than publish a table missing rows. It is stricter about one
 * thing than that function and looser about another, because it reads a real
 * mailbox rather than one caller's list: chunks are separated by delivery
 * attempt (chunk.total), and a repeated chunk index is the resend it always is,
 * not a fatal contradiction. Both are marked at the functions concerned.
 *
 * COMPLETELY SEPARATE from the license/telemetry script: different script,
 * different spreadsheet, different token — zero LockService contention with
 * the engineers' license checks.
 *
 * ONE-TIME SETUP (owner):
 *   1. Create the "ScrapeX Staging" spreadsheet.
 *   2. Extensions -> Apps Script -> paste this file.
 *   3. Project Settings -> Script Properties -> add FUNNEL_TOKEN = <long random>.
 *      Generate one, e.g. in PowerShell:
 *      -join ((48..57)+(97..122) | Get-Random -Count 48 | % {[char]$_})
 *   4. Deploy -> New deployment -> Web app -> Execute as: Me,
 *      Who has access: Anyone. Copy the /exec URL.
 *      "Anyone" is required: ScrapeX posts without a Google sign-in, so any
 *      narrower setting answers the login page instead of this script.
 *   5. Locally: set SCRAPEX_FUNNEL_URL + SCRAPEX_FUNNEL_TOKEN, run: scrapex funnel-test
 *   6. Reload the spreadsheet once: a "ScrapeX" menu appears, with
 *      "Rebuild tables from _INBOX" and the automatic-sync switches.
 *   7. OPTIONAL, once every producer you use is on a build that signs (any
 *      ScrapeX new enough to have pasted THIS file): Script Properties ->
 *      FUNNEL_REQUIRE_SIGNATURE = yes. See "REQUEST SIGNING" below for what
 *      that turns on and why it is not the default.
 * ========================================================================== */

const INBOX_SHEET = "_INBOX";
const RUNS_SHEET = "_RUNS";

const INBOX_HEADER = [
  "received_at", "source_key", "payload_version", "chunk_index", "chunk_total",
  "client", "scraped_at", "run_ref", "payload_json",
];
const RUNS_HEADER = [
  "received_at", "source_key", "client", "level", "message",
];

function doPost(e) {
  const lock = LockService.getScriptLock();
  if (!lock.tryLock(20000)) return jsonReply({ ok: false, error: "busy" });
  try {
    if (!e || !e.postData || !e.postData.contents) {
      return jsonReply({ ok: false, error: "empty request" });
    }
    const request = JSON.parse(e.postData.contents);
    const expected = PropertiesService.getScriptProperties().getProperty("FUNNEL_TOKEN");
    if (!expected || !constantTimeEquals_(String(request.token || ""), expected)) {
      return jsonReply({ ok: false, error: "unauthorized" });
    }
    const signatureProblem = signatureProblem_(request, expected);
    if (signatureProblem) return jsonReply({ ok: false, error: signatureProblem });

    if (request.action === "staging_chunk") return handleChunk(request.payload);
    if (request.action === "run_log") return handleRunLog(request);
    if (request.action === "staging_sync") return handleSyncNow();
    return jsonReply({ ok: false, error: "unknown action: " + request.action });
  } catch (error) {
    return jsonReply({ ok: false, error: String(error && error.message || error) });
  } finally {
    lock.releaseLock();
  }
}

/* ============================================================================
 * REQUEST SIGNING
 *
 * Every request ScrapeX sends carries `signature` = HMAC-SHA256, in lowercase
 * hex, over the canonical form of the request body with the signature field
 * removed, keyed with the SAME shared token (scrapex/funnel.py: canonical_body
 * + sign_body). The token still travels in the body and is still the thing that
 * authenticates; the signature proves the rows were not rewritten in flight.
 *
 * TWO deliberate softnesses, both about not failing shut:
 *
 *   1. A request WITHOUT a signature is accepted, unless the owner sets the
 *      script property FUNNEL_REQUIRE_SIGNATURE = yes. An owner running an
 *      older ScrapeX build (or an older copy of this script somewhere) must not
 *      have their sends start answering "unauthorized" with no explanation.
 *      Turn the property on once every producer is current; from then on an
 *      unsigned request is refused in words that say exactly that.
 *   2. A request WITH a signature is always verified, property or not — if the
 *      client bothered to prove itself, a mismatch is never waved through.
 *
 * What this is NOT: replay protection. No timestamp or nonce rides along, so a
 * captured request could be posted again — it arrives as a duplicate chunk,
 * which the assembler below drops (reassemble_ keeps the last copy per index).
 * Nothing here ever writes, echoes or returns the token or the signature: the
 * replies say "signature mismatch", never what was expected or received.
 * ========================================================================== */

const REQUIRE_SIGNATURE_PROPERTY = "FUNNEL_REQUIRE_SIGNATURE";

/* "" when the request may proceed, otherwise the refusal to send back. */
function signatureProblem_(request, token) {
  const required = String(PropertiesService.getScriptProperties()
    .getProperty(REQUIRE_SIGNATURE_PROPERTY) || "").toLowerCase() === "yes";
  const provided = request && request.signature ? String(request.signature) : "";
  if (!provided) {
    return required
      ? "signature required (FUNNEL_REQUIRE_SIGNATURE is on; update the sender)"
      : "";
  }
  const unsigned = {};
  Object.keys(request).forEach(function (key) {
    if (key !== "signature") unsigned[key] = request[key];
  });
  const expected = hexHmacSha256_(canonicalJson_(unsigned), token);
  return constantTimeEquals_(provided, expected) ? "" : "signature mismatch";
}

function hexHmacSha256_(text, key) {
  const bytes = Utilities.computeHmacSha256Signature(text, key);
  let hex = "";
  for (let i = 0; i < bytes.length; i++) {
    hex += ("0" + (bytes[i] & 0xff).toString(16)).slice(-2);
  }
  return hex;
}

/* The GAS half of funnel.canonical_body(): keys sorted, no whitespace, pure
 * ASCII. Written out rather than handed to JSON.stringify because stringify
 * keeps insertion order (which no transport preserves) and emits non-ASCII
 * verbatim (which would make the hashed bytes depend on a charset guess).
 *
 * The payload contract carries strings, small integers and null only, so this
 * mirrors Python's json.dumps(sort_keys=True, separators=(",", ":"),
 * ensure_ascii=True) exactly for everything that can actually arrive. */
function canonicalJson_(value) {
  if (value === null || typeof value !== "object") return asciiJson_(value);
  if (Array.isArray(value)) {
    const items = [];
    for (let i = 0; i < value.length; i++) items.push(canonicalJson_(value[i]));
    return "[" + items.join(",") + "]";
  }
  const keys = Object.keys(value).sort();
  const parts = [];
  for (let k = 0; k < keys.length; k++) {
    parts.push(asciiJson_(keys[k]) + ":" + canonicalJson_(value[keys[k]]));
  }
  return "{" + parts.join(",") + "}";
}

/* JSON.stringify already escapes exactly what Python escapes for ASCII input
 * (quote, backslash, the control characters); the replace finishes the job for
 * U+007F and up, which Python's ensure_ascii=True escapes and stringify does
 * not. Lowercase hex, four digits, surrogate pairs left as two escapes —
 * byte-for-byte what Python produces. */
function asciiJson_(value) {
  return JSON.stringify(value).replace(/[\u007f-\uffff]/g, function (ch) {
    return "\\u" + ("0000" + ch.charCodeAt(0).toString(16)).slice(-4);
  });
}

/* Compare every character, never short-circuit: a wrong signature must not be
 * discoverable one character at a time from how fast this answers. Apps Script
 * has no real constant-time primitive and the length is compared up front (all
 * signatures are the same length anyway), hence "-ish" — it closes the obvious
 * leak, it is not a hardware-grade guarantee. */
function constantTimeEquals_(a, b) {
  if (a.length !== b.length) return false;
  let diff = 0;
  for (let i = 0; i < a.length; i++) diff |= a.charCodeAt(i) ^ b.charCodeAt(i);
  return diff === 0;
}

/* Append one chunk row to _INBOX. Dumb by design: the payload cell is stored
 * verbatim; Python validates and reassembles at ingest. */
function handleChunk(payload) {
  if (!payload || !payload.source_key) return jsonReply({ ok: false, error: "missing payload" });
  const sheet = ensureSheet(INBOX_SHEET, INBOX_HEADER);
  const chunk = payload.chunk || { index: 1, total: 1 };
  sheet.appendRow([
    new Date(), payload.source_key, payload.payload_version, chunk.index, chunk.total,
    payload.client, payload.scraped_at, payload.run_ref || "", JSON.stringify(payload),
  ]);
  return jsonReply({ ok: true });
}

/* Append one log/alert row to _RUNS (producers report run outcomes here). */
function handleRunLog(request) {
  const sheet = ensureSheet(RUNS_SHEET, RUNS_HEADER);
  sheet.appendRow([
    new Date(), request.source_key || "", request.client || "",
    request.level || "info", request.message || "",
  ]);
  return jsonReply({ ok: true });
}

function ensureSheet(name, header) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    sheet.appendRow(header);
    sheet.setFrozenRows(1);
  }
  return sheet;
}

function jsonReply(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

/* Health check: open the /exec URL in a browser. */
function doGet() {
  return ContentService.createTextOutput("ScrapeX staging funnel operational");
}

/* ============================================================================
 * TABLE SYNC — one readable tab per source_key, rebuilt from _INBOX.
 *
 * _INBOX is a mailbox, not a table: one row per wire chunk, payload as JSON.
 * This turns the newest COMPLETE batch of each source into a plain tab named
 * after its source_key. Every run replaces the tab wholesale, so the tab is
 * always exactly "the last thing that source sent" — that is the sync.
 *
 * Nothing here mutates _INBOX. It stays the append-only audit log that Python
 * ingests; the tabs are derived and safe to delete at any time.
 * ========================================================================== */

/* Sources that should not get a tab of their own (self-test noise). */
const SYNC_SKIP_SOURCES = ["FUNNEL_SELFTEST"];

/* Mirrors gdrive.MAX_EXPORT_ROWS: a runaway batch must not try to paint
 * millions of cells and die halfway through, leaving a half-written tab. */
const SYNC_MAX_ROWS = 40000;

/* The payload contract this script speaks (payload.PAYLOAD_VERSION). A batch
 * announcing anything else is skipped loudly instead of parsed on a guess. */
const SYNC_PAYLOAD_VERSION = 1;

/* Cells are written as TEXT on purpose. Python already canonicalised every
 * value to a string (outputs._canonical_cell) so that 15 and 15.0 cannot fork
 * the record hash; letting Sheets re-type "1124.87" or "2026-07-21" would undo
 * exactly that. Flip to false if you would rather do arithmetic in the sheet
 * than keep the values byte-identical to what ScrapeX sent. */
const SYNC_AS_TEXT = true;

/* Menu. Appears after the spreadsheet is reloaded once. */
function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu("ScrapeX")
    .addItem("Rebuild tables from _INBOX", "syncTables")
    .addSeparator()
    .addItem("Sync automatically every 15 minutes", "installSyncTrigger")
    .addItem("Stop syncing automatically", "removeSyncTrigger")
    .addToUi();
}

/* Assemble now and SAY what happened. The Python side calls this right after
 * the last chunk and surfaces the reply in its sync UI — closing the loop
 * that used to end at "delivered": a delivery the assembler then refused
 * looked like success while the table sat stale. */
function handleSyncNow() {
  try {
    return jsonReply({ ok: true, report: syncTables() });
  } catch (error) {
    return jsonReply({ ok: false, error: String(error && error.message || error) });
  }
}

/* Rebuild every source's tab. Safe to run at any time, as often as you like. */
function syncTables() {
  // A DOCUMENT lock, deliberately NOT the script lock doPost uses: holding that
  // one here would make an arriving chunk wait, and the Python client treats a
  // "busy" ack as a permanent refusal (funnel.py: not retryable). Sync must
  // never be able to fail a live delivery. Concurrent syncs are still excluded
  // from each other, which is all this needs.
  const lock = LockService.getDocumentLock();
  if (!lock.tryLock(30000)) {
    throw new Error("another sync is already running — try again in a moment");
  }
  try {
    const report = rebuildTables_();
    logSync_(report);
    notifySync_(report);
    return report;
  } finally {
    lock.releaseLock();
  }
}

function rebuildTables_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const inbox = ss.getSheetByName(INBOX_SHEET);
  if (!inbox) return { written: [], skipped: [{ source: "(all)", reason: "no _INBOX sheet yet" }] };

  const batches = collectBatches_(inbox);
  const written = [];
  const skipped = [];

  Object.keys(batches).sort().forEach(function (sourceKey) {
    if (SYNC_SKIP_SOURCES.indexOf(sourceKey) !== -1) return;
    let batch;
    try {
      batch = newestCompleteBatch_(batches[sourceKey]);
    } catch (error) {
      skipped.push({ source: sourceKey, reason: String(error && error.message || error) });
      return;
    }
    try {
      // One source failing must not stop the others: a ragged batch from one
      // connector should never leave every other table stale.
      writeTable_(ss, sourceKey, batch.header, batch.rows);
      written.push({ source: sourceKey, rows: batch.rows.length, scraped_at: batch.scraped_at });
    } catch (error) {
      skipped.push({ source: sourceKey, reason: String(error && error.message || error) });
    }
  });
  return { written: written, skipped: skipped };
}

/* Read _INBOX once and group the chunks by (source_key, scraped_at) — the same
 * batch identity payload.reassemble_chunks() enforces — and then once more, by
 * chunk.total.
 *
 * That third level is one delivery ATTEMPT. When a batch is too large for a
 * single Apps Script execution the client re-plans it into smaller chunks and
 * resends the whole sequence (funnel.FunnelClient._deliver), so _INBOX can
 * legitimately hold an abandoned 3-chunk attempt beside the complete 6-chunk
 * one that followed it. Reassembling those together would count 8 chunks where
 * the envelope promises 6 and refuse a batch that did arrive whole.
 *
 * The envelope COLUMNS are for human eyes only; every value used here is read
 * back out of payload_json, because Sheets may re-type a cell (an ISO date in
 * particular) and a re-typed key would split one batch into two. */
function collectBatches_(inbox) {
  const values = inbox.getDataRange().getValues();
  const bySource = {};
  for (let i = 1; i < values.length; i++) {  // row 0 is INBOX_HEADER
    const raw = values[i][8];                // payload_json
    if (!raw) continue;
    let payload;
    try {
      payload = JSON.parse(raw);
    } catch (error) {
      continue;  // a corrupt cell must not take down every table
    }
    if (!payload || !payload.source_key) continue;
    const chunk = payload.chunk || { index: 1, total: 1 };
    const sourceKey = String(payload.source_key);
    const batchKey = String(payload.scraped_at || "");
    const attemptKey = String(Number(chunk.total || 1));
    if (!bySource[sourceKey]) bySource[sourceKey] = {};
    if (!bySource[sourceKey][batchKey]) bySource[sourceKey][batchKey] = {};
    if (!bySource[sourceKey][batchKey][attemptKey]) bySource[sourceKey][batchKey][attemptKey] = [];
    bySource[sourceKey][batchKey][attemptKey].push({
      index: Number(chunk.index || 1),
      total: Number(chunk.total || 1),
      version: Number(payload.payload_version),
      header: payload.header || [],
      rows: payload.rows || [],
      scraped_at: batchKey,
      row: i + 1,
    });
  }
  return bySource;
}

/* Pick the newest batch that is COMPLETE, and reassemble it.
 *
 * Newest-first, then fall back: a batch still arriving (chunk 3 of 5 landed a
 * second ago) must not blank the tab, and must not replace it with a partial
 * table either — the previous complete batch stays published until the new one
 * is whole. Silent truncation is the one outcome this must never produce. */
function newestCompleteBatch_(batchesForSource) {
  const keys = Object.keys(batchesForSource).sort().reverse();  // ISO8601 sorts chronologically
  const problems = [];
  for (let k = 0; k < keys.length; k++) {
    const attempts = batchesForSource[keys[k]];
    // Newest ATTEMPT first, by the last inbox row it wrote: when a batch was
    // re-planned smaller, the attempt that finished is the one whose chunks
    // landed last. An abandoned earlier attempt is then just inbox history.
    const totals = Object.keys(attempts).sort(function (a, b) {
      return lastInboxRow_(attempts[b]) - lastInboxRow_(attempts[a]);
    });
    for (let t = 0; t < totals.length; t++) {
      try {
        return reassemble_(attempts[totals[t]]);
      } catch (error) {
        problems.push(keys[k] + " (" + totals[t] + " chunks): " +
                      String(error && error.message || error));
      }
    }
  }
  throw new Error("no complete batch (" + problems.join("; ") + ")");
}

function lastInboxRow_(chunks) {
  let last = 0;
  for (let i = 0; i < chunks.length; i++) last = Math.max(last, chunks[i].row);
  return last;
}

/* The GAS mirror of payload.reassemble_chunks(). Deliberately as strict:
 * a MISSING chunk is a loud error, never silently-partial data.
 *
 * A REPEATED chunk is different, and is not an error: a delivery whose ack was
 * lost, or a spooled batch retried on a later run, sends the same index again
 * (funnel.py keeps the outbox entry until every piece is acked, so a resend is
 * the normal shape of recovery). The last copy to arrive wins, and the batch
 * publishes instead of sitting forever at "have 5, expected 3". */
function reassemble_(chunks) {
  const total = chunks[0].total;
  if (chunks[0].version !== SYNC_PAYLOAD_VERSION) {
    throw new Error("payload_version " + chunks[0].version + "; this script speaks " +
                    SYNC_PAYLOAD_VERSION);
  }
  const byIndex = {};
  for (let i = 0; i < chunks.length; i++) {
    if (chunks[i].total !== total) {
      throw new Error("chunks disagree on total (" + chunks[i].total + " vs " + total + ")");
    }
    byIndex[String(chunks[i].index)] = chunks[i];  // later inbox row overwrites
  }
  const ordered = Object.keys(byIndex).map(function (key) { return byIndex[key]; })
    .sort(function (a, b) { return a.index - b.index; });
  if (ordered.length !== total) {
    throw new Error("expected " + total + " chunks, have " + ordered.length);
  }
  let rows = [];
  for (let i = 0; i < ordered.length; i++) {
    if (ordered[i].index !== i + 1) {
      throw new Error("chunk sequence broken: expected " + (i + 1) + ", got " + ordered[i].index);
    }
    rows = rows.concat(ordered[i].rows);
  }
  return { header: ordered[0].header, rows: rows, scraped_at: ordered[0].scraped_at };
}

/* Replace one tab with header + rows. */
function writeTable_(ss, sourceKey, header, rows) {
  const tab = tabNameFor_(sourceKey);
  if (!header || !header.length) throw new Error("batch carries no header");
  if (rows.length > SYNC_MAX_ROWS) {
    throw new Error(rows.length + " rows exceeds SYNC_MAX_ROWS=" + SYNC_MAX_ROWS);
  }
  const width = header.length;
  for (let i = 0; i < rows.length; i++) {
    if (rows[i].length !== width) {
      // setValues() would reject the whole block anyway; saying which row is
      // ragged turns an opaque Sheets error into a connector bug report.
      throw new Error("row " + i + " has " + rows[i].length + " cells, header has " + width);
    }
  }

  let sheet = ss.getSheetByName(tab);
  if (!sheet) sheet = ss.insertSheet(tab);
  sheet.clear();
  const data = [header].concat(rows);
  const range = sheet.getRange(1, 1, data.length, width);
  if (SYNC_AS_TEXT) range.setNumberFormat("@");
  range.setValues(data);
  sheet.setFrozenRows(1);
  return sheet;
}

/* A tab named after the source, without ever colliding with the two sheets
 * this system actually depends on. */
function tabNameFor_(sourceKey) {
  const cleaned = String(sourceKey).replace(/[\[\]\*\/\\\?:]/g, "_").substring(0, 90).trim();
  const name = cleaned || "UNNAMED";
  if (name === INBOX_SHEET || name === RUNS_SHEET) {
    throw new Error("source_key '" + sourceKey + "' collides with a system sheet");
  }
  return name;
}

/* Record every sync in _RUNS. An automatic trigger fails invisibly otherwise:
 * without this line a broken 15-minute sync just looks like stale tabs. */
function logSync_(report) {
  const sheet = ensureSheet(RUNS_SHEET, RUNS_HEADER);
  const wrote = report.written.map(function (w) { return w.source + "(" + w.rows + ")"; });
  const level = report.skipped.length ? "warning" : "info";
  let message = "sync: wrote " + report.written.length + " tab(s)" +
                (wrote.length ? " — " + wrote.join(", ") : "");
  if (report.skipped.length) {
    message += " | skipped: " + report.skipped.map(function (s) {
      return s.source + " (" + s.reason + ")";
    }).join("; ");
  }
  sheet.appendRow([new Date(), "", "apps_script", level, message]);
}

/* Toast only when a human is watching; a time-driven run has no UI. */
function notifySync_(report) {
  try {
    SpreadsheetApp.getActiveSpreadsheet().toast(
      "Wrote " + report.written.length + " tab(s)" +
      (report.skipped.length ? ", skipped " + report.skipped.length : ""),
      "ScrapeX sync", 5);
  } catch (error) {
    // No UI attached (time-driven trigger). _RUNS already has the record.
  }
}

function installSyncTrigger() {
  removeSyncTrigger();
  ScriptApp.newTrigger("syncTables").timeBased().everyMinutes(15).create();
  notify_("Automatic sync is on: every 15 minutes.");
}

function removeSyncTrigger() {
  const triggers = ScriptApp.getProjectTriggers();
  for (let i = 0; i < triggers.length; i++) {
    if (triggers[i].getHandlerFunction() === "syncTables") ScriptApp.deleteTrigger(triggers[i]);
  }
}

function notify_(message) {
  try {
    SpreadsheetApp.getActiveSpreadsheet().toast(message, "ScrapeX", 5);
  } catch (error) {
    // Called from a context without a UI; nothing to show.
  }
}
